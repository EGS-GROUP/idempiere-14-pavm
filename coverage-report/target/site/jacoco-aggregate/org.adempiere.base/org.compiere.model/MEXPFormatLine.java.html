		StringBuilder sql = new StringBuilder("SELECT * ")

		ResultSet rs = null;

			pstmt = DB.prepareStatement (sql.toString(), trxName);

			throw e;
		} finally{
			DB.close(rs, pstmt);
			rs = null;
			pstmt = null;
		}
		
		return result;
	}
