package org.adempiere.process;

import java.sql.PreparedStatement;
import java.util.logging.Level;

import org.compiere.Adempiere;
import org.compiere.util.CLogMgt;
import org.compiere.util.CLogger;
import org.compiere.util.DB;

/**
 *	Sign Database Build
 *	[ 1851190 ] Running outdated client can cause data corruption
 *
 *  @author Carlos Ruiz
 */
public class SignDatabaseBuild
{
	/**	Static Logger	*/
	private static final CLogger	s_log	= CLogger.getCLogger (SignDatabaseBuild.class);

	public static void main(String[] args)
	{
		Adempiere.startupEnvironment(false);
		CLogMgt.setLevel(Level.FINE);
		s_log.info("Sign Database Build");
		s_log.info("-------------------");
		
		if (! DB.isConnected()) {
			s_log.info("No DB Connection");
			System.exit(1);
		}
		
		PreparedStatement updateStmt = null;
		try {
			String upd = "UPDATE AD_System SET LastBuildInfo = ?";
			updateStmt = DB.prepareStatement(upd, null);
			updateStmt.setString(1, Adempiere.getImplementationVersion());
			s_log.info(upd);
			System.out.println(upd);  // Also show the update to the console
			updateStmt.executeUpdate();

		} catch (Exception ex) {
			ex.printStackTrace();
		} finally {
			DB.close(updateStmt);
		}

	}

}	//	SignDatabaseBuild
