package org.adempiere.webui.component;

/**
 * Static method to create new grid layout
 * @author Low Heng Sin
 */
public class GridFactory {

	/**
	 * New instance of grid for form layout
	 * @return Grid
	 */
	public static Grid newGridLayout() {
		Grid grid = new Grid();
		grid.setSclass("grid-layout");
		return grid;
	}
}
