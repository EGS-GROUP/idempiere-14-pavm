package fi.jawsy.jawwa.zk.atmosphere;

/**
 * class to hold two value, left and right
 * @author hengsin
 *
 * @param <L>
 * @param <R>
 */
public class Either<L, R> {
	private L left;
	private R right;
	
	/**
	 * @param l left value
	 * @param r right value
	 */
	public Either(L l, R r) {
		left = l;
		right = r;
	}
	
	/**
	 * @return left value
	 */
	public L getLeftValue() {
		return left;
	}
	
	/**
	 * @return right value
	 */
	public R getRightValue() {
		return right;
	}
}
